#include <stdio.h>
#define MAXFILA     26
#define MAXCADFILA 100

main()
{
    int sala[MAXFILA][MAXCADFILA];
    int qtdfila = 0;
    int tamMaiorFila = 0;
    int i, j, qa;
    char letras[MAXFILA];

    for (i = 0; i < MAXFILA; i++) {
        letras[i] = 65 + i;     // 65 == 'A' do alfabeto. ;-)
    }
    /* Agora temos um vetor de caracteres (NÃO STRING), onde letras[0] tem
     * o 'A', letras[1] tem o 'B' e assim sucessivamente.
     * Garanto-mos ter uma maneira bem mais divertida de se fazer isso.
     * Conversamos em algoritmos III ;-)
     * Vi alguns grupos realmente criando uma string com strncpy.
     * Outros usando uma função com um extenso switch/case
     * TODOS VÁLIDOS (se estiverem corretos, claro)
     * */

    printf("Quantas filas tem esta sala (%i MAXIMO)? ", MAXFILA);
    scanf("%i", &qtdfila);
    /* Não ficou especificado o que deve ser feito se colocar mais do que o 
     * limite. Então aceito qualquer solução (menos deixar ler mais):
     * a - ficar lendo em um laço até ser válido
     * b - assumir o máximo se a valor foi digitado errado
     * c - encerrar o programa, pois o usuário não respeitou um limite
     * Aqui eu optei em usar o b
     * */
    if ((qtdfila <= 0) || (qtdfila > MAXFILA)) {
        printf("Valor %i invalido. Assumindo %i\n", qtdfila, MAXFILA);
        qtdfila = MAXFILA;
    }

    printf("Quantos assentos tem sua maior fila (%i MAXIMO)? ", MAXCADFILA);
    scanf("%i", &tamMaiorFila);
    if ((tamMaiorFila <= 0) || (tamMaiorFila > MAXCADFILA)) {
        printf("Valor %i invalido. Assumindo %i\n", tamMaiorFila, MAXCADFILA);
        tamMaiorFila = MAXCADFILA;
    }

    /* Inicializando a matriz: etapa A. SEM USAR FUNÇÃO. Veja outra resposta
     * em breve publicada usando função.
     * */
    for (i = 0; i < qtdfila; i++) {
        printf("Quantos assentos tem a fila %c (1 a %i)?", letras[i], tamMaiorFila);
        scanf("%i", &qa);
        if ((qa <= 0) || (qa > tamMaiorFila)) {
            printf("Valor %i invalido. Considerando %i\n", qa, tamMaiorFila);
        }
        for (j = 0; j < qa; j++) {
            sala[i][j] = 0;
        }
        for (j = qa; j < tamMaiorFila; j++) {
            sala[i][j] = -1;
        }
    }

    /* Um print simples para depuracao (não faz parte do trabalho,
     * deve ser removido depois. Não coloca O ou L. Este for
     * é apenas para testar a conclusão da etapa A)
     * */
    printf("\n");
    for (i = 0; i < qtdfila; i++) {
        printf("%c = ", letras[i]);
        for (j = 0; j < tamMaiorFila; j++) {
            printf("%2d ", sala[i][j]);
        }
        printf("\n");
    }
}
